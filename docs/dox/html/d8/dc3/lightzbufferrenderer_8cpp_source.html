124 Vec3 toBarycenteric(const Vec3 &p, const Vec3 &a, const Vec3 &b, const Vec3 &c)
125 {
126  Vec3 v0 = b - a, v1 = c - a, v2 = p - a;
127  float d00 = v0.dot(v0);
128  float d01 = v0.dot(v1);
129  float d11 = v1.dot(v1);
130  float d20 = v2.dot(v0);
131  float d21 = v2.dot(v1);
132  float denom = d00 * d11 - d01 * d01;
133  float v = (d11 * d20 - d01 * d21) / denom;
134  float w = (d00 * d21 - d01 * d20) / denom;
135  float u = 1.0f - v - w;
136  return Vec3(u, v, w);
137 }
