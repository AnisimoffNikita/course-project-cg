35  int32 width = image.getWidth();

46  {

48

50  {
51  for (int32 j = 0; j < kernelSize; j++)
52  {
53  pixelPosX = (x + j - kernelSize / 2 + width) % width;
54  pixelPosY = (y + i - kernelSize / 2 + height) % height;
55  uint8 color = saved.at(pixelPosY, pixelPosX);
56  float gxVal = gx[i][j];
57  float gyVal = gy[i][j];
58  resultX += color * gxVal;
59  resultY += color * gyVal;
60  }
61  }
62 
63  result = fabs(resultX) + fabs(resultY);
64  float angle;
65 
66  if (resultX == 0 && resultY == 0)
67  {
68  angle = 0;
69  }
70  else if (resultX == 0)
71  {
72  angle = 90;
73  }
74  else

77  }
78

80  {

82  }
